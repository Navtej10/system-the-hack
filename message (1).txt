const INITIAL_POLICIES = [
  { id: 'reskilling', title: 'Invest in AI Reskilling', description: 'Direct capital towards retraining the workforce for the automated economy.', impacts: { employment: 15, governmentBudget: -20, publicHappiness: 10 }, riskLevel: 'Low' },
  { id: 'ubi', title: 'Introduce Universal Basic Income', description: 'Stabilize household consumption amid automation-induced job displacement.', impacts: { inequality: -25, publicHappiness: 20, economy: 5, governmentBudget: -30 }, riskLevel: 'High' },
  { id: 'tax_corp', title: 'Tax AI Corporations', description: 'Rebalance wealth by taxing productivity gains from algorithmic labour.', impacts: { governmentBudget: 25, inequality: -10, economy: -15, employment: -5 }, riskLevel: 'Medium' },
  { id: 'restrict_ai', title: 'Restrict Automation', description: 'Mandate human presence in critical sectors to protect societal cohesion.', impacts: { employment: 20, economy: -20, publicHappiness: 5 }, riskLevel: 'High' },
  { id: 'promote_ai', title: 'Promote AI Innovation', description: 'Accelerate the transition to a high-efficiency autonomous economy.', impacts: { economy: 25, inequality: 15, employment: -15, governmentBudget: -10 }, riskLevel: 'Medium' }
];

const API_BASE = 'http://localhost:3001';
const DECISION_HISTORY_SIZE = 5;

const METRIC_DEFS = [
  { key: 'employment', label: 'Employment', icon: '👥' },
  { key: 'economy', label: 'Economy', icon: '📈' },
  { key: 'publicHappiness', label: 'Happiness', icon: '😊' },
  { key: 'inequality', label: 'Equality', icon: '⚖️' },
  { key: 'governmentBudget', label: 'Budget', icon: '💰' }
];

let gameState = { turn: 0, metrics: {}, status: 'playing' };
let currentPolicies = [...INITIAL_POLICIES];
let decisionHistory = [];

function applyPolicyLogic(state, policy) {
  const newMetrics = { ...state.metrics };
  for (const [key, value] of Object.entries(policy.impacts)) newMetrics[key] += value;
  
  if (newMetrics.economy > 80) newMetrics.employment += 5;
  if (newMetrics.inequality > 70) newMetrics.publicHappiness -= 10;
  if (newMetrics.employment < 30) newMetrics.publicHappiness -= 15;
  if (newMetrics.publicHappiness < 40) newMetrics.economy -= 10;
  if (newMetrics.governmentBudget < 20) newMetrics.economy -= 5;

  newMetrics.publicHappiness -= 1; newMetrics.governmentBudget -= 1;
  for (const k of Object.keys(newMetrics)) newMetrics[k] = Math.min(Math.max(newMetrics[k] + (Math.floor(Math.random()*5)-2), 0), 100);

  const vals = Object.values(newMetrics);
  let status = 'playing';
  if (vals.filter(v => v < 20).length >= 2) status = 'lost';
  else if (vals.every(v => v > 75)) status = 'won';
  else if (state.turn + 1 >= 10) status = 'lost';

  return { turn: state.turn + 1, metrics: newMetrics, status };
}

function renderMetrics() {
  const grid = document.getElementById('metrics-grid'); grid.innerHTML = '';
  METRIC_DEFS.forEach(def => {
    const val = gameState.metrics[def.key];
    const color = (val >= 70) ? 'var(--success)' : (val >= 40) ? 'var(--warning)' : 'var(--danger)';
    grid.innerHTML += `<div class="metric-box"><div class="metric-header"><span class="metric-label">${def.label}</span><span class="metric-icon">${def.icon}</span></div><div class="metric-val" style="color:${color}">${val}</div><div class="metric-bar-bg"><div class="metric-bar" style="width:${val}%;background:${color}"></div></div></div>`;
  });
  grid.innerHTML += `<div class="target-panel"><div style="font-family:var(--font-mono);font-size:10px;color:var(--accent);font-weight:bold;margin-bottom:5px;">TARGET PARAMETER</div><div style="font-size:14px;font-weight:500;">Win: All Sectors > 75%</div></div>`;
  document.getElementById('ui-turn-display').textContent = `${gameState.turn} / 10`;
  const isCrit = Object.values(gameState.metrics).some(v => v < 30);
  document.getElementById('ui-global-status').innerHTML = isCrit ? '<span style="color:var(--warning)">UNSTABLE</span>' : '<span style="color:var(--success)">STABLE</span>';
}

function renderPolicies(showLoading = false) {
  const grid = document.getElementById('policies-grid');
  grid.innerHTML = '';
  if (showLoading) {
    grid.innerHTML = '<div class="policies-loading">Predictive Engine generating next interventions…</div>';
    return;
  }
  const disabled = gameState.status !== 'playing';
  currentPolicies.forEach(p => {
    const safeId = p.id.replace(/'/g, "\\'");
    grid.innerHTML += `<div class="policy-card ${disabled?'disabled':''}" onclick="onPolicyClick('${safeId}')"><div class="policy-header"><div class="policy-title">${p.title}</div><div class="policy-risk risk-${p.riskLevel}">${p.riskLevel}</div></div><div class="policy-desc">${p.description}</div><div class="policy-impacts">${Object.keys(p.impacts).map(k=>`<span class="impact-tag">#${k}</span>`).join('')}</div></div>`;
  });
}

async function onPolicyClick(id) {
  if (gameState.status !== 'playing') return;
  const policy = currentPolicies.find(p => p.id === id);
  if (!policy) return;

  decisionHistory.push({ title: policy.title, turn: gameState.turn });
  if (decisionHistory.length > DECISION_HISTORY_SIZE) decisionHistory.shift();

  gameState = applyPolicyLogic(gameState, policy);
  const logType = (gameState.status==='won')?'success':(gameState.status==='lost')?'failure':'';
  const feed = document.getElementById('ai-feed');
  feed.innerHTML += `<div class="log-entry ${logType}"><div class="log-turn">LOG // TURN ${gameState.turn}</div>Intervention ${policy.title} deployed. Analytical models confirm state shift.</div>`;
  feed.scrollTop = feed.scrollHeight;
  renderMetrics();

  if (gameState.status !== 'playing') {
    renderPolicies();
    setTimeout(() => {
      document.getElementById('game-overlay').style.display = 'flex';
      document.getElementById('overlay-icon').textContent = gameState.status==='won'?'🕊️':'📉';
      document.getElementById('overlay-title').textContent = gameState.status==='won'?'SOCIETAL EQUILIBRIUM':'SYSTEM COLLAPSE';
      document.getElementById('overlay-desc').textContent = gameState.status==='won'?'Mission Success. Equilibrium achieved.':'Simulation Terminated. Governance framework compromised.';
    }, 600);
    return;
  }

  renderPolicies(true);
  try {
    const res = await fetch(`${API_BASE}/api/generate-interventions`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        chosenPolicy: { title: policy.title, description: policy.description },
        metrics: gameState.metrics,
        turn: gameState.turn,
        decisionHistory
      })
    });
    const data = await res.json();
    if (data.interventions && data.interventions.length > 0) {
      currentPolicies = data.interventions;
      feed.innerHTML += `<div class="log-entry system"><div class="log-turn">LOG // TURN ${gameState.turn}</div>Predictive Engine: ${data.interventions.length} new interventions generated (butterfly effect chain).</div>`;
    } else {
      currentPolicies = [...INITIAL_POLICIES];
      feed.innerHTML += `<div class="log-entry failure"><div class="log-turn">LOG // TURN ${gameState.turn}</div>API fallback: reverting to default interventions.</div>`;
    }
  } catch (err) {
    currentPolicies = [...INITIAL_POLICIES];
    feed.innerHTML += `<div class="log-entry failure"><div class="log-turn">LOG // TURN ${gameState.turn}</div>Predictive Engine offline. Fallback interventions loaded.</div>`;
  }
  feed.scrollTop = feed.scrollHeight;
  renderPolicies();
}

function initGame() {
  document.getElementById('game-overlay').style.display = 'none';
  gameState = { turn: 0, metrics: { employment:45, economy:46, publicHappiness:48, inequality:50, governmentBudget:52 }, status: 'playing' };
  currentPolicies = [...INITIAL_POLICIES];
  decisionHistory = [];
  document.getElementById('ai-feed').innerHTML = '<div class="log-entry system"><div class="log-turn">LOG // TURN 0</div>Simulation initialized. System awaiting directives.</div>';
  renderMetrics(); renderPolicies();
}

initGame();

const EDGE_COLOR  = 0xe2f0f8;
const FACE_COLOR  = 0x0b1828;
const WATER_COLOR = 0x0a1520;
const RAIL_COLOR  = 0xb0d0e0;

function edgeMat(opa = 0.95) { return new THREE.LineBasicMaterial({ color: EDGE_COLOR, transparent: true, opacity: opa }); }
function faceMat(opa = 0.92) { return new THREE.MeshBasicMaterial({ color: FACE_COLOR, transparent: true, opacity: opa, side: THREE.DoubleSide }); }

function building(w, h, d, x, y, z, edgeOpa = 0.95) {
  const geo = new THREE.BoxGeometry(w, h, d);
  const mesh = new THREE.Mesh(geo, faceMat()); mesh.position.set(x, y + h / 2, z); scene.add(mesh);
  const edges = new THREE.EdgesGeometry(geo);
  const lines = new THREE.LineSegments(edges, edgeMat(edgeOpa)); lines.position.copy(mesh.position); scene.add(lines);
}

function flat(w, d, x, z, y = 0, color = FACE_COLOR, opa = 1) {
  const mesh = new THREE.Mesh(new THREE.PlaneGeometry(w, d), new THREE.MeshBasicMaterial({ color, transparent: true, opacity: opa, side: THREE.DoubleSide }));
  mesh.rotation.x = -Math.PI / 2; mesh.position.set(x, y, z); scene.add(mesh);
}

function polyLine(points, color = EDGE_COLOR, opa = 0.5, y = 0.3) {
  const pts = points.map(([x, z]) => new THREE.Vector3(x, y, z));
  scene.add(new THREE.Line(new THREE.BufferGeometry().setFromPoints(pts), new THREE.LineBasicMaterial({ color, transparent: true, opacity: opa })));
